```python
def _build_article(
    self,
    pr: PRRecord,
    comment: CommentRecord,
    classified: ClassifiedComment,
) -> str | None:
    """Build a KB article with Claude-synthesized body (D-01, D-02, D-04)."""
    sections = _CATEGORY_SECTIONS.get(classified.category, _CATEGORY_SECTIONS["other"])

    user_prompt = (
        f"PR Title: {pr.title}\n\n"
        f"Category: {classified.category}\n\n"
        f"Comment:\n{comment.body[:10000]}\n\n"
        f"Write the article using these sections:\n{sections}"
    )

    try:
        response = self._client.messages.create(
            model=self._model,
            max_tokens=1024,
            temperature=0.3,
            system=SYNTHESIS_SYSTEM_PROMPT,
            messages=[{"role": "user", "content": user_prompt}],
        )
    except anthropic.APIError as exc:
        logger.warning("API error synthesizing comment %d: %s", comment.comment_id, exc)
        self._failed.append({
            "file": f"comment-{comment.comment_id}",
            "reason": type(exc).__name__,
            "detail": str(exc),
        })
        return None

    synthesized_body = response.content[0].text

    frontmatter_lines = [
        _FRONTMATTER_DELIMITER,
        f"pr_url: {pr.url}",
        f"pr_title: {_yaml_str(pr.title)}",
        f"comment_url: {comment.url}",
        f"author: {_yaml_str(comment.author)}",
        f"date: {comment.created_at.isoformat()}",
        f"category: {classified.category}",
        f"confidence: {classified.confidence}",
        f"needs_review: {str(classified.needs_review).lower()}",
        f"comment_id: {comment.comment_id}",
        _FRONTMATTER_DELIMITER,
    ]

    body_parts = [
        "\n".join(frontmatter_lines),
        "",
        f"# {classified.summary}",
        "",
        synthesized_body,
    ]

    if comment.diff_hunk:
        body_parts.extend(["", "```", comment.diff_hunk, "```"])

    return "\n".join(body_parts) + "\n"
```

```python
def _reset_kb(self) -> None:
    """Clear manifest and delete all category article directories for regeneration."""
    self._manifest = {}
    self._save_manifest()
    # Delete category subdirectories (not INDEX.md or .manifest.json)
    if self._kb_dir.exists():
        for item in self._kb_dir.iterdir():
            if item.is_dir():
                import shutil
                shutil.rmtree(item)
```

```python
def generate_all(self, regenerate: bool = False) -> GenerateResult:
    """Read all classified-pr-N.json files, write new articles, update manifest."""
    self._written = 0
    self._skipped = 0
    self._filtered = 0
    self._failed = []
    self._category_slugs = {}

    if regenerate:
        self._reset_kb()

    for classified_path in self._find_classified_files():
        self._process_classified_file(classified_path)

    self._save_manifest()
    self._generate_index()
    return GenerateResult(
        written=self._written,
        skipped=self._skipped,
        filtered=self._filtered,
        failed=self._failed,
    )
```
